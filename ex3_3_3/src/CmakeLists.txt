cmake_minimum_required(VERSION 3.4)
PROJECT(ex3_3_3)

FIND_PACKAGE(ITK)
IF (ITK_FOUND )
  INCLUDE( ${ITK_USE_FILE} )
ENDIF(ITK_FOUND)

FIND_PACKAGE(VTK)

IF(VTK_FOUND)
  INCLUDE(${VTK_USE_FILE})
ENDIF(VTK_FOUND)

INCLUDE_DIRECTORIES(${ex3_3_3_SOURCE_DIR})

ADD_EXECUTABLE(ex3_3_3 ex3_3_3.cxx)

TARGET_LINK_LIBRARIES(ex3_3_3
 ${ITK_LIBRARIES}
 ${VTK_LIBRARIES}
)
